```js function getTaipeiDateParts(offsetDays) { const nowUTC = new Date(); const taipeiTime = new Date( nowUTC.getTime() + (8 * 60 * 60 * 1000) ); taipeiTime.setDate( taipeiTime.getDate() + (offsetDays || 0) ); const y = taipeiTime.getUTCFullYear(); const m = String( taipeiTime.getUTCMonth() + 1 ).padStart(2, "0"); const d = String( taipeiTime.getUTCDate() ).padStart(2, "0"); const WEEK = ["日","一","二","三","四","五","六"]; const w = WEEK[taipeiTime.getUTCDay()]; return { fileDate: y + "-" + m + "-" + d, label: y + "/" + Number(m) + "/" + Number(d) + "（" + w + "）" }; } ```
